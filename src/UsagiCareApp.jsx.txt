import React, { useState, useEffect } from "react";
import { initializeApp } from "firebase/app";
import { getDatabase, ref, onValue, set } from "firebase/database";
import { Card, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Avatar } from "@/components/ui/avatar";
import { motion } from "framer-motion";

const personalities = ["元気いっぱい", "おっとり", "甘えんぼう", "好奇心旺盛", "慎重派"];

const firebaseConfig = {
  apiKey: "YOUR_API_KEY",
  authDomain: "YOUR_PROJECT_ID.firebaseapp.com",
  databaseURL: "https://YOUR_PROJECT_ID.firebaseio.com",
  projectId: "YOUR_PROJECT_ID",
  storageBucket: "YOUR_PROJECT_ID.appspot.com",
  messagingSenderId: "SENDER_ID",
  appId: "APP_ID"
};

const app = initializeApp(firebaseConfig);
const database = getDatabase(app);

export default function UsagiCareApp() {
  const personalityQuotes = {
    "元気いっぱい": "今日も元気いっぱいだよ！",
    "おっとり": "のんびり過ごそうね。",
    "甘えんぼう": "もっとなでてほしいな♡",
    "好奇心旺盛": "あちこち探検したい！",
    "慎重派": "ちょっと様子を見てからね。"
  };

  const [personality, setPersonality] = useState(personalities[Math.floor(Math.random() * personalities.length)]);
  const [name, setName] = useState("");
  const [partnerName, setPartnerName] = useState("");
  const [careCount, setCareCount] = useState(0);
  const [isSleeping, setIsSleeping] = useState(false);
  const [careTime, setCareTime] = useState(0);
  const [lastReset, setLastReset] = useState(new Date().toDateString());
  const [feedCount, setFeedCount] = useState(0);
  const [careLog, setCareLog] = useState([]);
  const [outfit, setOutfit] = useState("default");
  const [usagiName, setUsagiName] = useState("");
  const [usagiGender, setUsagiGender] = useState("");

  useEffect(() => {
    if (new Date().toDateString() !== lastReset) {
      setCareTime(0);
      setLastReset(new Date().toDateString());
    }
    const now = new Date();
    setIsSleeping(now.getHours() >= 23);

    const dbRef = ref(database, "usagiData");
    onValue(dbRef, (snapshot) => {
      const data = snapshot.val();
      if (data) {
        setCareCount(data.careCount);
        setFeedCount(data.feedCount);
        setOutfit(data.outfit);
        setPersonality(data.personality);
        setCareLog(data.careLog || []);
      }
    });
  }, [lastReset]);

  const handleCare = () => {
    if (isSleeping) return alert("今は夜なのでウサギは寝ています");
    if (new Date().toDateString() !== lastReset) {
      setCareTime(0);
      setLastReset(new Date().toDateString());
    }
    if (careTime >= 60) return alert("1日の上限1時間に達しました");
    const newCareCount = careCount + 1;
    const newCareLog = [...careLog, `🐰 なでた (${new Date().toLocaleString()})`];
    setCareCount(newCareCount);
    setCareTime(careTime + 5);
    if (careCount % 5 === 4) {
      const newPersonality = personalities[Math.floor(Math.random() * personalities.length)];
      setPersonality(newPersonality);
    }
    setCareLog(newCareLog);
    set(ref(database, "usagiData"), {
      careCount: newCareCount,
      feedCount,
      outfit,
      personality,
      careLog: newCareLog
    });
  };

  const handleFeed = () => {
    setFeedCount(feedCount + 1);
    setCareLog([...careLog, `🥕 餌をあげた (${new Date().toLocaleString()})`]);
  };

  const changeOutfit = (newOutfit) => {
    setOutfit(newOutfit);
    setCareLog([...careLog, `🎀 着せ替え: ${newOutfit} (${new Date().toLocaleString()})`]);
  };

  const roomImage = isSleeping ? "/night-room.png" : "/day-room.png";

  return (
    <div className="flex flex-col items-center justify-center min-h-screen bg-pink-50 p-4" style={{ backgroundImage: `url(${roomImage})`, backgroundSize: 'cover' }}>
      <h1 className="text-2xl font-bold mb-4">🐰 ウサギのお世話アプリ</h1>
      <Card className="w-full max-w-sm">
        <CardContent className="flex flex-col items-center gap-4 p-4">
          <Avatar className="w-32 h-32 bg-white border-4 border-pink-300">
            <img src={`/${outfit}-${personality}-usagi-icon.png`} alt="うさぎ" className="w-full h-full object-cover rounded-full" />
          </Avatar>
          <Input placeholder="ウサギの名前を入力" value={usagiName} onChange={(e) => setUsagiName(e.target.value)} />
          <Input placeholder="ウサギの性別 (例: 男の子 / 女の子)" value={usagiGender} onChange={(e) => setUsagiGender(e.target.value)} />
          <Input placeholder="あなたの名前を入力" value={name} onChange={(e) => setName(e.target.value)} />
          <Input placeholder="パートナーの名前を入力" value={partnerName} onChange={(e) => setPartnerName(e.target.value)} />
          <motion.div whileTap={{ scale: 0.9 }} className="text-center">
            <p className="text-lg">{usagiName ? `${usagiName} (${usagiGender})` : (name && partnerName ? `${name} さんと ${partnerName} さんのウサギ` : "ウサギ")}</p>
            <p className="text-sm text-gray-500">性格: {personality}</p>
            <p className="text-sm italic text-pink-500">{personalityQuotes[personality]}</p>
            <p className="text-sm text-gray-500">今日のお世話時間: {careTime}分 / 60分</p>
            <p className="text-sm text-gray-500">なでた回数: {careCount}</p>
            <p className="text-sm text-gray-500">餌をあげた回数: {feedCount}</p>
          </motion.div>
          <Button onClick={handleCare} className="bg-pink-300 hover:bg-pink-400 text-white">ウサギをなでる</Button>
          <Button onClick={handleFeed} className="bg-green-300 hover:bg-green-400 text-white">ウサギに餌をあげる</Button>
          <div className="flex gap-2 mt-4">
            <Button onClick={() => changeOutfit("default")} className="bg-blue-300 hover:bg-blue-400 text-white">通常</Button>
            <Button onClick={() => changeOutfit("hat")} className="bg-yellow-300 hover:bg-yellow-400 text-white">帽子</Button>
            <Button onClick={() => changeOutfit("scarf")} className="bg-purple-300 hover:bg-purple-400 text-white">マフラー</Button>
          </div>
        </CardContent>
      </Card>
      <div className="w-full max-w-sm mt-4">
        <Card>
          <CardContent className="p-4">
            <h2 className="text-lg font-semibold mb-2">📋 お世話の記録</h2>
            <ul className="space-y-1">
              {careLog.length > 0 ? (
                careLog.map((log, index) => <li key={index} className="text-sm">{log}</li>)
              ) : (
                <li className="text-sm text-gray-500">まだ記録はありません。</li>
              )}
            </ul>
          </CardContent>
        </Card>
      </div>
    </div>
  );
}
